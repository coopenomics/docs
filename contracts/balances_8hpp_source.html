130void add_balance(eosio::name source, eosio::name username, eosio::asset quantity,
131 eosio::name contract) {
132 // Если баланс не найден, создаем новую запись.
133 // В противном случае, увеличиваем существующий баланс.
134
135
136 eosio::check(username != ""_n, "В поле memo должен быть указан получатель баланса");
137 balances_index balances(source, username.value);
138
139 auto balances_by_contract_and_symbol =
140 balances.template get_index<"byconsym"_n>();
141 auto contract_and_symbol_index =
142 combine_ids(contract.value, quantity.symbol.code().raw());
143
144 auto balance =
145 balances_by_contract_and_symbol.find(contract_and_symbol_index);
146
147 if (balance == balances_by_contract_and_symbol.end()) {
148 balances.emplace(source, [&](auto &b) {
149 b.id = balances.available_primary_key();
150 b.contract = contract;
151 b.quantity = quantity;
152 });
153 } else {
154 balances_by_contract_and_symbol.modify(
155 balance, source, [&](auto &b) { b.quantity += quantity; });
156 };
157}

160void sub_balance(eosio::name source, eosio::name username, eosio::asset quantity,
161 eosio::name contract) {
162 // Если после вычитания баланс равен нулю, удаляем запись.
163 // В противном случае, уменьшаем существующий баланс.
164
165 balances_index balances(source, username.value);
166
167 auto balances_by_contract_and_symbol =
168 balances.template get_index<"byconsym"_n>();
169 auto contract_and_symbol_index =
170 combine_ids(contract.value, quantity.symbol.code().raw());
171
172 auto balance =
173 balances_by_contract_and_symbol.find(contract_and_symbol_index);
174
175 eosio::check(balance != balances_by_contract_and_symbol.end(),
176 "Баланс не найден");
177
178 eosio::check(balance->quantity >= quantity, "Недостаточный баланс");
179
180 if (balance->quantity == quantity) {
181
182 balances_by_contract_and_symbol.erase(balance);
183
184 } else {
185
186 balances_by_contract_and_symbol.modify(
187 balance, source, [&](auto &b) { b.quantity -= quantity; });
188 }
189}
