41 template<typename E, typename F>
42 static inline auto has_field( F flags, E field )
43 -> std::enable_if_t< std::is_integral_v<F> && std::is_unsigned_v<F> &&
44 std::is_enum_v<E> && std::is_same_v< F, std::underlying_type_t<E> >, bool>
45 {
46 return ( (flags & static_cast<F>(field)) != 0 );
47 }
48
49 template<typename E, typename F>
50 static inline auto set_field( F flags, E field, bool value = true )
51 -> std::enable_if_t< std::is_integral_v<F> && std::is_unsigned_v<F> &&
52 std::is_enum_v<E> && std::is_same_v< F, std::underlying_type_t<E> >, F >
53 {
54 if( value )
55 return ( flags | static_cast<F>(field) );
56 else
57 return ( flags & ~static_cast<F>(field) );
58 }

183 eosio::block_signing_authority get_producer_authority()const {
184 if( producer_authority.has_value() ) {
185 bool zero_threshold = std::visit( [](auto&& auth ) -> bool {
186 return (auth.threshold == 0);
187 }, *producer_authority );
188 // zero_threshold could be true despite the validation done in regproducer2 because the v1.9.0 eosio.system
189 // contract has a bug which may have modified the producer table such that the producer_authority field
190 // contains a default constructed eosio::block_signing_authority (which has a 0 threshold and so is invalid).
191 if( !zero_threshold ) return *producer_authority;
192 }
193 return convert_to_block_signing_authority( producer_key );
194 }

196 // The unregprod and claimrewards actions modify unrelated fields of the producers table and under the default
197 // serialization behavior they would increase the size of the serialized table if the producer_authority field
198 // was not already present. This is acceptable (though not necessarily desired) because those two actions require
199 // the authority of the producer who pays for the table rows.
200 // However, the rmvproducer action and the onblock transaction would also modify the producer table in a similar
201 // way and increasing its serialized size is not acceptable in that context.
202 // So, a custom serialization is defined to handle the binary_extension producer_authority
203 // field in the desired way. (Note: v1.9.0 did not have this custom serialization behavior.)
204
205 template<typename DataStream>
206 friend DataStream& operator << ( DataStream& ds, const producer_info& t ) {
207 ds << t.owner
208 << t.total_votes
209 << t.producer_key
210 << t.is_active
211 << t.url
212 << t.unpaid_blocks
213 << t.last_claim_time
214 << t.location;
215
216 if( !t.producer_authority.has_value() ) return ds;
217
218 return ds << t.producer_authority;
219 }
220
221 template<typename DataStream>
222 friend DataStream& operator >> ( DataStream& ds, producer_info& t ) {
223 return ds >> t.owner
224 >> t.total_votes
225 >> t.producer_key
226 >> t.is_active
227 >> t.url
228 >> t.unpaid_blocks
229 >> t.last_claim_time
230 >> t.location
231 >> t.producer_authority;
232 }
233 };
